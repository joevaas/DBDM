Aim

To develop an ER model for Agriculture Management System and perform EER to Object Database (ODB) mapping with proper keys and relationships.

Algorithm
Identify entities (Farmer, Farm, Crop)
Define attributes for each entity
Establish relationships (owns, grows)
Apply EER concepts (specialization: Crop → FoodCrop, CashCrop)
Map entities to tables (ODB mapping)
Define Primary Key & Foreign Key
Insert sample data
Display output

Code (Short & Full)
-- Farmer Table
CREATE TABLE Farmer(
    farmer_id INT PRIMARY KEY,
    name VARCHAR(30)
);

-- Farm Table
CREATE TABLE Farm(
    farm_id INT PRIMARY KEY,
    location VARCHAR(30),
    farmer_id INT,
    FOREIGN KEY(farmer_id) REFERENCES Farmer(farmer_id)
);

-- Crop (Parent)
CREATE TABLE Crop(
    crop_id INT PRIMARY KEY,
    name VARCHAR(30)
);

-- FoodCrop (Child - EER)
CREATE TABLE FoodCrop(
    crop_id INT PRIMARY KEY,
    nutrition VARCHAR(30),
    FOREIGN KEY(crop_id) REFERENCES Crop(crop_id)
);

-- CashCrop (Child - EER)
CREATE TABLE CashCrop(
    crop_id INT PRIMARY KEY,
    price INT,
    FOREIGN KEY(crop_id) REFERENCES Crop(crop_id)
);

-- Relationship (Farm grows Crop)
CREATE TABLE Farm_Crop(
    farm_id INT,
    crop_id INT,
    PRIMARY KEY(farm_id,crop_id),
    FOREIGN KEY(farm_id) REFERENCES Farm(farm_id),
    FOREIGN KEY(crop_id) REFERENCES Crop(crop_id)
);

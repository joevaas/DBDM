Aim

To create a database with Trigger, Function, and Sequence for order processing.

Algorithm
Create tables: CUSTOMER, PRODUCT, CUST_ORDER, ORDER_PRODUCT
Create SEQUENCE for PCODE
Create TRIGGER to allow only Pen, Eraser, Pencil
Create FUNCTION to calculate total cost
Insert data and test
Code (Full SQL + PL/SQL)
-- Tables
CREATE TABLE CUSTOMER(
    CID INT PRIMARY KEY,
    NAME VARCHAR(30)
);

CREATE TABLE PRODUCT(
    PCODE INT PRIMARY KEY,
    PNAME VARCHAR(30),
    UNIT_PRICE INT
);

CREATE TABLE CUST_ORDER(
    OCODE INT PRIMARY KEY,
    ODATE DATE,
    CID INT,
    FOREIGN KEY(CID) REFERENCES CUSTOMER(CID)
);

CREATE TABLE ORDER_PRODUCT(
    OCODE INT,
    PCODE INT,
    QTY INT,
    FOREIGN KEY(OCODE) REFERENCES CUST_ORDER(OCODE),
    FOREIGN KEY(PCODE) REFERENCES PRODUCT(PCODE)
);

-- iii. Sequence for PCODE
CREATE SEQUENCE prod_seq
START WITH 1
INCREMENT BY 1;

-- Trigger (only Pen, Eraser, Pencil allowed)
CREATE OR REPLACE TRIGGER trg_product
BEFORE INSERT ON PRODUCT
FOR EACH ROW
BEGIN
    IF :NEW.PNAME NOT IN ('Pen','Eraser','Pencil') THEN
        RAISE_APPLICATION_ERROR(-20001,'Invalid Product');
    END IF;
END;
/

-- Insert using Sequence
INSERT INTO PRODUCT VALUES (prod_seq.NEXTVAL,'Pen',10);
INSERT INTO PRODUCT VALUES (prod_seq.NEXTVAL,'Eraser',5);
INSERT INTO PRODUCT VALUES (prod_seq.NEXTVAL,'Pencil',8);

-- Function to calculate total cost
CREATE OR REPLACE FUNCTION total_cost(p_order INT)
RETURN NUMBER
IS
    total NUMBER;
BEGIN
    SELECT SUM(P.UNIT_PRICE * O.QTY)
    INTO total
    FROM PRODUCT P
    JOIN ORDER_PRODUCT O ON P.PCODE = O.PCODE
    WHERE O.OCODE = p_order;

    RETURN total;
END;
/

-- Sample Data
INSERT INTO CUSTOMER VALUES (1,'Ravi');

INSERT INTO ORDER_PRODUCT VALUES (101,1,2);

-- Call Function
SELECT total_cost(101) FROM dual;
Output
Trigger restricts invalid products
Sequence generates PCODE automatically

Result
